def platform_keyboard():

@dp.message(Command("start"))
async def start(message: types.Message):

        reply_markup=platform_keyboard()
    )

@dp.callback_query(F.data.in_(['android', 'ios']))
async def platform(call: types.CallbackQuery):
    user_choices[call.from_user.id] = {"platform": call.data}
    await call.message.edit_text(
        "🚀 Теперь выбери, для чего тебе нужен ВПН:",
        reply_markup=purpose_keyboard()
    )
    await call.answer()

@dp.callback_query(F.data.in_(['games', 'social', 'all', 'advice']))
async def purpose(call: types.CallbackQuery):
    platform = user_choices.get(call.from_user.id, {}).get("platform", "android")
    app = APPS[platform][call.data]
    
    await call.message.edit_text(
        f"✅ {app['name']}\n\n👉 [Скачать]({app['url']})\n\n✨ Надеюсь, поможет!",
        reply_markup=None
    )
    await call.answer()
